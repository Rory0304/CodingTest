## N과 M 기본 로직


### permutations 라이브러리
```python
from itertools import permutations
from sys import stdin
n, m = map(int, stdin.read().split())

n_list = permutations([i for i in range(1, n+1)], m)

for n in n_list:
    print(*n, sep=" ")
```


### 재귀 함수
```python
from sys import stdin
N, M = map(int, stdin.read().split())

def makeNumber(n):
    if len(n) != M:
        for i in range(1, N+1):
            if i not in n:
                temp = n[::]
                temp.append(i)
                makeNumber(temp)
    else:
        print(" ".join(list(map(str, n))))

for i in range(1, N+1):
    makeNumber([i])
```


## N과 M(4)
```python
from sys import stdin
N, M = map(int, stdin.read().split())

def makeNumber(n):
    if len(n) != M:
        for i in range(1, N+1):
            if n[-1] <= i:
                temp = n[::]
                temp.append(i)
                makeNumber(temp)
    else:
        print(*n, sep=" ")

for i in range(1, N+1):
    makeNumber([i])
    
```
